112void Menu::writeMonthStats(std::ofstream& file, const BST<std::string>& tree, const std::map<std::string, WeatherLog>& dataMap, int year, int month) {
113 WeatherLog data = getRecordsByYearMonth(tree, dataMap, year, month);
114 if(data.GetSize() == 0) return;
115
116 Vector<float> wind = extractWindSpeeds(data), temp = extractTemperatures(data), solar = extractSolarRadiation(data);
117 float wAvg = mean(wind)*3.6f, wStdev = stdev(wind)*3.6f, wMad = 0;
118 float tAvg = mean(temp), tStdev = stdev(temp), tMad = 0;
119 float solarTotal = calculateTotalSolar(solar);
120
121 //compute MAD (Mean Absolute Deviation, 1 decimal) for wind
122 float wMean = mean(wind)*3.6f;
123 int w_n = 0;
124 for(int i=0; i<wind.GetSize(); i++) {
125 if(!std::isnan(wind[i])) { wMad += std::abs(wind[i]*3.6f - wMean); w_n++; }
126 }
127 if(w_n>0) wMad /= w_n;
128
129 //compute MAD for temp
130 float tMean = mean(temp);
131 int t_n = 0;
132 for(int i=0; i<temp.GetSize(); i++) {
133 if(!std::isnan(temp[i])) { tMad += std::abs(temp[i] - tMean); t_n++; }
134 }
135 if(t_n>0) tMad /= t_n;
136
137 file << monthName(month) << ",";
138 if(!std::isnan(wAvg))
139 file << std::fixed << std::setprecision(1)
140 << wAvg << "(" << wStdev << ", " << wMad << "),";
141 else
142 file << " ,";
143 if(!std::isnan(tAvg))
144 file << std::fixed << std::setprecision(1)
145 << tAvg << "(" << tStdev << ", " << tMad << "),";
146 else
147 file << " ,";
148 if(!std::isnan(solarTotal))
149 file << std::fixed << std::setprecision(1) << solarTotal;
150 else
151 file << " ";
152 file << "\n";
153}
