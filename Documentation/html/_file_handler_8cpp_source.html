140//process one CSV line for weather data
141void FileHandler::processCSVLine(const std::string& line, const std::map<std::string, int>& colMap,
142 BST<std::string>& dateTree, std::map<std::string, WeatherLog>& dataMap) {
143 std::stringstream ss(line);
144 std::string cell;
145 Vector<std::string> fields;
146 while (std::getline(ss, cell, ',')) {
147 fields.pushBack(cell);
148 }
149
150 //extract and parse fields.
151 //Expects day/month/year for date, hh:mm for time.
152 try {
153 std::string dateTimeStr = fields[colMap.at("WAST")];
154 // If CSV entry is "15/07/2025 09:45"
155 std::stringstream dts(dateTimeStr);
156 std::string datePart, timePart;
157 std::getline(dts, datePart, ' ');
158 std::getline(dts, timePart);
159
160 Date date = parseDate(datePart);
161 MyTime time = parseTime(timePart);
162
163 float wind = fields[colMap.at("S")].empty() ? NAN : std::stof(fields[colMap.at("S")]);
164 float temp = fields[colMap.at("T")].empty() ? NAN : std::stof(fields[colMap.at("T")]);
165 float solar = fields[colMap.at("SR")].empty() ? NAN : std::stof(fields[colMap.at("SR")]);
166
167 WeatherEntry w {date, time, wind, temp, solar};
168 //key is year month as string
169 std::string key = std::to_string(date.GetYear()) + "-" + (date.GetMonth() < 10 ? "0" : "") + std::to_string(date.GetMonth());
170
171 dataMap[key].pushBack(w);
172 if (!dateTree.search(key)) {
173 dateTree.insert(key);
174 }
175 } catch (...) {
176 //skip any problematic lines
177 }
178}
