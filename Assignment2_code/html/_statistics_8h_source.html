41  *
42  * Loops through the numbers and divides by how many were valid (not NaN).
43  * Gives NaN if the vector is empty or totally full of NaNs.
44  *

51  float sum = 0.0f; int n = 0;
52  for(int i=0;i<data.GetSize();i++) {
53  float v = data[i];
54  if(!std::isnan(v)) { sum += v; n++; }

61  * @brief Computes the sample standard deviation for your vector (skips NaN values).
62  *
63  * Useful for showing the "spread" of your data.
64  * Returns NaN if there are fewer than 2 valid numbers.
65  *
66  * @tparam T Numeric type in the vector.

70 template<typename T>
71 float stdev(const Vector<T>& data) {
72  int n = 0; float avg = mean(data), sum2 = 0.0f;
73  if(std::isnan(avg)) return NAN;
74  for(int i=0;i<data.GetSize();i++) {
75  float v = data[i];
76  if(!std::isnan(v)) { sum2 += (v - avg)*(v - avg); n++; }
77  }
78  return (n<=1) ? NAN : std::sqrt(sum2/(n-1));
79 }

83  * @brief Calculates Pearson correlation coefficient between two vectors.
84  *
85  * Measures how closely the two datasets change together (from -1 to 1).
86  * Any value that's NaN in either vector will be ignored for the calculation.
87  * Throws an invalid_argument if the vectors are different lengths or empty.
88  * Returns NaN if there aren't enough valid pairs to calculate, and rounds the result to two decimals.
89  *
90  * @tparam T Numeric type (float).
91  * @param x First vector.
92  * @param y Second vector.
93  * @return Correlation coefficient (rounded to 2 decimals), or NaN/0 for special cases.
94  * @throws std::invalid_argument if vector sizes don't match.
95  */
96 template<typename T>
97 float pearson(const Vector<T>& x, const Vector<T>& y) {
98  if(x.GetSize() != y.GetSize() || x.GetSize() == 0)
99  throw std::invalid_argument("Vector dimensions mismatch");
100  float sumX=0, sumY=0, sumXY=0, sumX2=0, sumY2=0;
101  int n=0;
102  for(int i=0;i<x.GetSize();i++) {
103  float xv=x[i], yv=y[i];
104  if(std::isnan(xv) || std::isnan(yv)) continue;
105  sumX += xv;
106  sumY += yv;
107  sumXY += xv * yv;
108  sumX2 += xv * xv;
109  sumY2 += yv * yv;
110  n++;
111  }
112  if(n < 2) return NAN;
113  float num = sumXY - (sumX*sumY)/n;
114  float denom = std::sqrt((sumX2-sumX*sumX/n)*(sumY2-sumY*sumY/n));
115  return denom == 0 ? 0.0f : RoundVal(num/denom, 2);
116 }
